Overview

This project analyzes tweets from Donald Trump’s time in office using text mining, regular expressions, and sentiment analysis. The analysis focuses on common words, repeated phrases, emotional language, and changes in tweet style over time.


```{r}
#| include: false
#regular expression, str, making data set able to do text analysis 
#converted to lowercase, removed links, and common stopwords.
great_stop_words <- stop_words|>
  filter(word != "great")

tidy_tweet <- realDonaldTrump_in_office |>
  select(Time, Tweet.Text) |>
  mutate(
    line = row_number(),
    Tweet.Text = str_to_lower(Tweet.Text),
    Tweet.Text = str_remove_all(Tweet.Text, "https\\S+")) |> #regular expression 1, remove text with https with any nonwhitespace character after it
  unnest_tokens(word, Tweet.Text) |>
  filter(!word %in% c("rt", "amp", "trump", "realdonaldtrump")) |> #don't include these words
  anti_join(great_stop_words, by = "word")
```

```{r}
#| include: FALSE
nrc_sentiments <- get_sentiments("nrc")
bing_sentiments <- get_sentiments("bing")
smart_stopwords <- get_stopwords(source = "smart")
```

```{r}

#Regular expressions
realDonaldTrump_in_office |>
  mutate(fake_word = str_extract(Tweet.Text, regex("\\bfake\\b", ignore_case = TRUE))) |>

realDonaldTrump_in_office |>
  mutate(fake_phrase = str_extract(Tweet.Text, regex("\\bfake\\s+\\w+", ignore_case = TRUE))) |> #regular expression 2, including the word fake with a word after it
  count(fake_phrase, sort = TRUE) |>
  slice_max(n, n = 15)
```
